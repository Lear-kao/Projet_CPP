struct card_gen
{
    type : bool
    d_o_a (dead or alive) : bool
    cost : int
    name : string
}

struct card_comb : card_gen
{
    class : int
    strength : int
    toughness : int
    capacity : list[string]
}

struct cart_inst : card_gen
{
    class : int
    t_capacity : int
    n_stack : int
}

struct heap_card
{
    number_card : int

    func pick_one -> card_gen
    func add_one -> void
    func shuffle -> void
    func cut (*p1 : heap_card, *p2 : heap_card) -> void
}

struct player
{
    life : int
    number_card_hand : int
    deck : heap_card : int
    power : int
    nb_player : bool
    name : string
}

struct game
{
    l_player[2]
    n_turn
    base_l_card : heap_card
    cemetary: heap_card
}
